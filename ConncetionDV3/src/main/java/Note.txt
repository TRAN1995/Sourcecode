CREATE OR REPLACE PROCEDURE INSERT_OR_UPDATE_TABLETABLE(
 P_ID IN NUMBER,
 P_CODE IN VARCHAR2,
 P_STATUS IN VARCHAR2,
 P_RESPONSE OUT VARCHAR2

) AS
BEGIN
  IF P_ID = 0 THEN
     INSERT INTO tttd_lich_su (ma_yeu_cau,trang_thai)
     VALUES(P_CODE,P_STATUS);
     P_RESPONSE :='Insert operation successful';
  ELSE
    UPDATE tttd_lich_su
    SET ma_yeu_cau= P_CODE, trang_thai=P_STATUS
    WHERE id=P_ID;
    P_RESPONSE :='UPDATE operation successful';
  END IF;
EXCEPTION
  WHEN OTHERS THEN
    P_RESPONSE := 'Error: ' || SQLERRM;
END INSERT_OR_UPDATE_TABLETABLE;
